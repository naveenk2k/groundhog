# Groundhog companion dependencies.
#
# This file is intentionally minimal for now (issue #1: server boots and
# authenticates over localhost). Later issues add:
#   - yt-dlp (transcript fetching, #2)
#   - sentence-transformers (embeddings, #3)
#   - sqlite-vec (corpus storage, #3)
#   - an Anthropic client (scoring, #5)
fastapi
uvicorn

# Local embeddings (#3): CPU-friendly, keeps everything on-machine.
sentence-transformers

# Corpus storage (#3): sqlite-vec is loaded as a runtime SQLite extension.
sqlite-vec

# apsw instead of the stdlib sqlite3 module (#3): Apple's system Python
# (what a bare `python3 -m venv` in install.sh picks up unless Homebrew/
# python.org Python is on PATH first) ships a SQLite build without
# loadable-extension support, so sqlite3.Connection.enable_load_extension
# doesn't exist there. apsw bundles its own SQLite build with extension
# loading enabled and works the same on every platform.
apsw
